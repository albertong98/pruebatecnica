CREATE TABLE TPEDIDO (
order_id VARCHAR(100) NOT NULL,
uuid VARCHAR(200) NOT NULL,
order_priority VARCHAR(1) NOT NULL,
order_date DATE NOT NULL,
region VARCHAR(50),
country VARCHAR(50),
item_type VARCHAR(50),
sales_channel VARCHAR(50),
ship_date DATE,
units_sold INTEGER,
unit_price DOUBLE,
unit_cost DOUBLE,
total_revenue DOUBLE,
total_cost DOUBLE,
total_profit DOUBLE,
self_link VARCHAR(200),
PRIMARY KEY(uuid)
)
